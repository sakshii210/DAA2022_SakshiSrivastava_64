Q1. #include<bits/stdc++.h>
using namespace std;
long matChainOrder(int *p,int n)      {
    int m[n][n];
    int i,j,k,l,q;
    for(i=1;i<n;i++)
    m[i][i]=0;
    for(l=2;l<n;l++)
    {
        for(i=1;i<n-l+1;i++)
        {
            j=i+l-1;
            m[i][j]=INT_MAX;
            for(k=i;k<=j-1;k++)
            {
                q=m[i][k]+m[k+1][j]+p[i-1]*p[k]*p[j];
                if(q<m[i][j])
                m[i][j]=q;
            }
        }
    }
    return m[1][n-1];
}
int main()
{
    int n;
    cin>>n;
    int p[n+1];
    for(int i=0;i<n;i++)
    {
        cin>>p[i]>>p[i+1];
    }
    cout<<matChainOrder(p,n+1);
    return 0;
}

Q3.#include<bits/stdc++.h>
using namespace std;
int main()  {
    int n;
    cin>>n;
    int i,j,a[n];
    for(i=0;i<n;i++)
    cin>>a[i];
    int sum=0;
    for(i=0;i<n;i++)
    sum+=a[i];
    if(sum%2!=0)
    {
        cout<<"no";
        return 0;
    }
    sum=sum/2;
    bool s[n+1][sum+1];
    for(i=0;i<=n;i++)
    {
        for(j=0;j<=sum;j++)
        {
            if(j==0)
            s[i][j]=1;
            else if(i==0)
            s[i][j]=0;
            else
            {
                if(a[i-1]>j)
                s[i][j]=s[i-1][j];
                else
                s[i][j]=(s[i-1][j] || s[i-1][j-a[i-1]]);
            }
        }
    }
    if(s[n][sum])
    cout<<"yes";
    else
    cout<<"no";
    return 0;
}
